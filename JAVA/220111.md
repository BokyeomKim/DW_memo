package study;

public class Be {

	public static void main(String[] args) {
	
		
		int 국어 = 100;
		int 수학 = 50;
		int 영어 = 88;
		int 과학 = 38;
		final int 과목수 = 4;// 상수의미
		int sum = 0; // 총합 
		double avg = 0;// 평균
		//총합점수 평균점수
		
		sum=(국어+수학+영어+과학);
		avg=(국어+수학+영어+과학)/과목수;
		
		System.out.println("총합 :" +sum);
		System.out.println("평균 :"+avg) ;
	}

}
